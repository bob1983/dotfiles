```ruby
# 'C' means Ctrl
# 'M' means Alt
# 'Super' means Win

when 'C', 'Ctrl'
  mask |= X11::ControlMask
when 'M', 'Alt'
  mask |= X11::Mod1Mask
when 'Super', 'Win'
  mask |= X11::Mod4Mask
when 'Shift'
  mask |= X11::ShiftMask
end
```
